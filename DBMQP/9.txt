   create user 'emp3'@'%' identified by 'unknownuser';

3. GRANT select on employee to 'emp2'@'localhost' with grant option;
   GRANT update(Empname, deptno) on employee to 'emp2'@'localhost' with grant option;

4. sudo /opt/lampp/bin/mysql -u emp2 -p
   GRANT select on employee to 'emp1'@'localhost';
   GRANT update(Empname, deptno) on employee to 'emp1'@'localhost';
   
5. sudo /opt/lampp/bin/mysql -u root -p
   grant all privileges on *.* to 'emp3'@'%' with grant option;

6. show grants for 'emp1'@'localhost';
